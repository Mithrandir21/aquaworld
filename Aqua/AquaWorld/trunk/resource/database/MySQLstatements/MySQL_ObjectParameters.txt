CREATE TABLE IF NOT EXISTS `ObjectParameters` (
  `ObjectParametersID` int(11) NOT NULL AUTO_INCREMENT,
  `SalinityLow` double NOT NULL,
  `SalinityHigh` double NOT NULL,
  `PHLow` double NOT NULL,
  `PHHigh` double NOT NULL,
  `GHLow` double NOT NULL,
  `GHHigh` double NOT NULL,
  `TemperatureLow` double NOT NULL,
  `TemperatureHigh` double NOT NULL,
  `KHLow` double DEFAULT NULL,
  `KHHigh` double DEFAULT NULL,
  `MagnesiumLow` double DEFAULT NULL,
  `MagnesiumHigh` double DEFAULT NULL,
  `CalciumLow` double DEFAULT NULL,
  `CalciumHigh` double DEFAULT NULL,
  `SpaceNeeded` double DEFAULT NULL,
  `OthersSizeLow` double DEFAULT NULL,
  `OthersSizeHigh` double DEFAULT NULL,
  PRIMARY KEY (`ObjectParametersID`)
) ENGINE=InnoDB COMMENT='The Parameters for a any given Fish, Coral or Invertebrate.';
