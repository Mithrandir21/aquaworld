CREATE TABLE IF NOT EXISTS `FishObject` (
  `FishObjectID` int(11) NOT NULL AUTO_INCREMENT,
  `PopularName` text,
  `Species` text NOT NULL,
  `Genus` text,
  `Description` text,
  `Size` double NOT NULL,
  `ObjectParametersID` int(11) NOT NULL,
  `FishExclusionID` int(11) NOT NULL,
  PRIMARY KEY (`FishObjectID`),
  CONSTRAINT `FishExclusion` FOREIGN KEY `FishExclusion`(`FishExclusionID`) REFERENCES `FishExclusionList` (`FishExclusionID`) ON DELETE CASCADE,
  CONSTRAINT `FishParameters` FOREIGN KEY `FishParameters`(`ObjectParametersID`) REFERENCES `ObjectParameters` (`ObjectParametersID`) ON DELETE CASCADE  
) ENGINE=InnoDB ROW_FORMAT=DYNAMIC AUTO_INCREMENT=1 ;
