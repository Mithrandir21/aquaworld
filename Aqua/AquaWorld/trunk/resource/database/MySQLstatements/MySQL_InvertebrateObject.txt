CREATE TABLE IF NOT EXISTS `InvertebrateObject` (
  `InvertebrateObjectID` int(11) NOT NULL AUTO_INCREMENT,
  `PopularName` text,
  `Species` text NOT NULL,
  `Genus` text,
  `Description` text,
  `InvertebratesType` text NOT NULL,
  `ObjectParametersID` int(11) NOT NULL,
  `InvertebrateExclusionID` int(11) NOT NULL,
  PRIMARY KEY (`InvertebrateObjectID`),
  CONSTRAINT `InvertebrateParameters` FOREIGN KEY `InvertebrateParameters`(`ObjectParametersID`) REFERENCES `ObjectParameters` (`ObjectParametersID`),
  CONSTRAINT `InvertebrateExclusions` FOREIGN KEY `InvertebrateExclusions`(`InvertebrateExclusionID`) REFERENCES `InvertebrateExclusionList` (`InvertebrateExclusionID`)
) ENGINE=InnoDB ROW_FORMAT=DYNAMIC AUTO_INCREMENT=1 ;
